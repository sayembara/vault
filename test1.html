/** * SANRAT Race Point Calculator - RECTIFIED * Optimized for TBNB/Point precision and robust tiering. */ function calculateRaceResults(participants, config = {}) { // 1. Configuration with Defaults const { basePoints = 1000, decayFactor = 0.95, minPoints = 1, topCarsPerTeam = 2, // Thresholds ordered DESCENDING for proper 'break' logic multiplierThresholds = [ { min: 3001, multiplier: 2.0 }, { min: 201, max: 3000, multiplier: 1.5 }, { min: 61, max: 200, multiplier: 1.0 }, { min: 1, max: 60, multiplier: 0.5 } ] } = config; // 2. Input Validation if (!Array.isArray(participants)) throw new Error('Participants must be an array'); if (participants.length === 0) { return { metadata: { totalCars: 0 }, individualResults: [], constructorResults: [] }; } // Validate fields and check for unique ranks const rankCheck = new Set(); participants.forEach((car, index) => { if (!car.carId || !car.teamId || typeof car.rank !== 'number') { throw new Error(`Invalid data at index ${index}. Ensure carId, teamId, and rank (number) exist.`); } if (rankCheck.has(car.rank)) throw new Error(`Duplicate rank detected: ${car.rank}`); rankCheck.add(car.rank); }); const totalCars = participants.length; // 3. Determine Multiplier let multiplier = 0.5; for (const threshold of multiplierThresholds) { if (totalCars >= threshold.min && (!threshold.max || totalCars <= threshold.max)) { multiplier = threshold.multiplier; break; } } // 4. Individual Points Calculation const processedCars = participants.map(car => { let points; // The "Points Zone" is the top 50% of the field const pointsZoneLimit = Math.ceil(totalCars / 2); if (car.rank > pointsZoneLimit) { points = minPoints; } else { // Formula: base * decay^(rank-1) * multiplier const rawPoints = basePoints * Math.pow(decayFactor, car.rank - 1) * multiplier; // Use Math.round to avoid floating point issues in point totals points = Math.max(minPoints, Math.round(rawPoints)); } return { ...car, pointsEarned: points, multiplierUsed: multiplier, inPointsZone: car.rank <= pointsZoneLimit }; }).sort((a, b) => a.rank - b.rank); // 5. Constructor (Team) Logic const teamGroups = {}; processedCars.forEach(car => { if (!teamGroups[car.teamId]) teamGroups[car.teamId] = []; teamGroups[car.teamId].push(car); }); const constructorResults = Object.entries(teamGroups).map(([teamId, cars]) => { // Sort team's cars by points descending const sortedTeamCars = [...cars].sort((a, b) => b.pointsEarned - a.pointsEarned); // Take top N (e.g., top 2) const countingCars = sortedTeamCars.slice(0, topCarsPerTeam); const totalTeamPoints = countingCars.reduce((sum, c) => sum + c.pointsEarned, 0); return { teamId, totalTeamPoints, totalCarsInTeam: cars.length, countingCars: countingCars.map(c => ({ carId: c.carId, points: c.pointsEarned, rank: c.rank })) }; }).sort((a, b) => b.totalTeamPoints - a.totalTeamPoints); return { metadata: { totalCars, multiplierUsed: multiplier, timestamp: new Date().toISOString(), config: { basePoints, decayFactor, topCarsPerTeam } }, individualResults: processedCars, constructorResults: constructorResults.map((team, i) => ({ ...team, position: i + 1 })) }; } /** * Utility for Testing Display */ function formatResults(res) { if (res.metadata.totalCars === 0) return "Empty Race."; let report = `\n--- SANRAT RACE REPORT [${res.metadata.multiplierUsed}x Multiplier] ---\n`; report += `Top Individual: ${res.individualResults[0].carId} (${res.individualResults[0].pointsEarned} pts)\n`; report += `Top Constructor: ${res.constructorResults[0].teamId} (${res.constructorResults[0].totalTeamPoints} pts)\n`; report += `--------------------------------------\n`; return report; } // --- TEST SUITE --- const mockRace = [ { carId: "ACE", teamId: "RED_BULL", rank: 1 }, { carId: "FLY", teamId: "RED_BULL", rank: 3 }, { carId: "ZEN", teamId: "FERRARI", rank: 2 }, { carId: "MAX", teamId: "FERRARI", rank: 4 }, { carId: "BOT", teamId: "MERCEDES", rank: 5 } ]; const results = calculateRaceResults(mockRace, { basePoints: 1000, decayFactor: 0.90 }); console.log(formatResults(results)); console.log("Detailed Constructor Data:", JSON.stringify(results.constructorResults, null, 2));
